\section*{A degree lower bound for domination-free cross-ratio hypergraphs}

\subsection*{1. Setup and definitions}

\paragraph{Cross-ratio.} For four \emph{distinct} points $a,b,c,d\in\mathbb P^1=
\mathbb C\cup\{\infty\}$ put
\[
\chi(a,b,c,d)=\frac{(a-c)(b-d)}{(a-d)(b-c)}\in\mathbb C\setminus\{0,1\},
\]
with the usual conventions if one entry is $\infty$. $\chi$ is invariant under the
diagonal action of $\mathrm{PGL}_2(\mathbb C)$ on the four points.

\paragraph{Hypergraph and cross-ratio degree.} Let $H=(V,E)$ be a \emph{$4$-uniform}
hypergraph: $V$ finite, each $e\in E$ a $4$-element subset of $V$, and $|E|=|V|-3$.
Fix an ordering of the four vertices of each edge (the degree below is independent of
this choice). A \emph{configuration} is an injective map $x\colon V\to\mathbb P^1$; for
a \emph{target} $\lambda=(\lambda_e)_{e\in E}$ it is a \emph{$\lambda$-solution} if
$\chi(x|_e)=\lambda_e$ for every $e$. For generic $\lambda$ the set of $\lambda$-solutions
modulo $\mathrm{PGL}_2$ is finite of constant size; this size is the \emph{cross-ratio
degree} $\deg(H)\in\mathbb Z_{\ge 0}$. (Concretely: pin three vertices to $0,1,\infty$
and count solutions of the remaining $|V|-3$ equations in the $|V|-3$ unknowns, for
generic $\lambda$.)

\paragraph{Cerberus / positivity (assumable fact).} $H$ is \emph{Cerberus} if for every
nonempty $F\subseteq E$, $\bigl|\bigcup_{e\in F}e\bigr|\ge|F|+3$. We may use:
$\deg(H)\ge 1$ iff $H$ is Cerberus, and $\deg(H)=0$ otherwise. (Cerberus is equivalent
to \emph{$(1,3)$-tightness}: $|E|=|V|-3$, and every $X\subseteq V$ with $|X|\ge3$ spans
at most $|X|-3$ edges.)

\paragraph{Domination.} For $v\in V$ write $F_v=\{e\in E: v\in e\}$ for the \emph{star}
of $v$ (the set of edges containing $v$). A vertex $u$ is \emph{dominated} if there is a
vertex $w\ne u$ with
\[
F_u\subseteq F_w,
\]
i.e.\ every edge through $u$ also passes through $w$. We call $H$ \emph{domination-free}
if no vertex is dominated, that is, no star is contained in another.

\subsection*{2. Conjecture to prove}

\begin{quote}
\textbf{Conjecture.} Every domination-free Cerberus hypergraph $H$ satisfies
\[
\boxed{\;\deg(H)\ \ge\ 2.\;}
\]
Equivalently (contrapositive): if $\deg(H)=1$ then $H$ has a dominated vertex.
\end{quote}

A counterexample --- a domination-free Cerberus $H$ with $\deg(H)=1$ --- would also
resolve the problem.

\subsection*{3. Context (true statements; provided to orient, not to constrain the proof)}

\textbf{C1 (Goldner's recursion, a purely combinatorial computation of $\deg$).}
If $|E|\le1$ then $\deg(H)=1$. Otherwise pick any edge $e_0=\{v_1,v_2,v_3,v_4\}$ and sum
over \emph{admissible splits}:
\[
\deg(H)=\sum_{W\ \mathrm{admissible}}\ \deg(H'_W)\cdot\deg(H''_W),
\]
where $W\subseteq V$ is admissible iff (A1) $v_1,v_2\in W$ and $v_3,v_4\notin W$;
(A2) $|W\cap e|\ne2$ for every $e\in E\setminus\{e_0\}$; and (A3)
$\#\{e\ne e_0:|e\cap W|\ge3\}=|W|-2$. Here $H'_W$ has vertex set $W\cup\{\ast\}$ and the
edges with $\ge3$ vertices in $W$ (their unique exterior vertex relabeled $\ast$);
$H''_W$ is the mirror image on $(V\setminus W)\cup\{\ast\}$. By (A2) every edge other
than $e_0$ goes to exactly one side, and by (A3) each side again satisfies
``$|E|=|V|-3$''.

\textbf{C2 (cancellation-free lower bound).} Every term of C1 is a product of
non-negative integers. Hence
\[
\deg(H)\ \ge\ \#\{\,W\ \mathrm{admissible}\ :\ H'_W\ \text{and}\ H''_W\ \text{are both Cerberus}\,\},
\]
since each such term is $\ge1$. In particular, exhibiting \emph{two} admissible splits
whose two sides are all Cerberus already gives $\deg(H)\ge2$. (This reduces the
conjecture to a statement about the existence of Cerberus splits; it is recorded only to
orient, and no particular choice of $e_0$ or $W$ is prescribed.)

\textbf{C3 (where the content lies).} A vertex $u$ of degree $1$ in $H$ (a \emph{leaf},
$|F_u|=1$) is automatically dominated, since its single edge contains three other
vertices $w$, each with $F_u\subseteq F_w$. Likewise if all edges share a common vertex
(a \emph{cone}) then $\deg(H)=1$ and that vertex dominates every other. Thus the
conjecture is trivial unless $H$ is leaf-free and not a cone; the substance is the
leaf-free, non-cone case.

\textbf{C4 (reduction hypothesis, optional).} One may additionally assume $H$ has no
proper subset $E'\subsetneq E$ that is itself square and Cerberus of degree $1$ (``no
proper tight degree-$1$ subsystem''): such a subsystem can be handled separately and
removed, so the residual hard case is the one with no such subsystem. This hypothesis
may be assumed or ignored; it is not required for the statement.

\textbf{C5 (evidence).} The conjecture has been checked exactly against the combinatorial
degree (C1) on large datasets: \emph{no} domination-free Cerberus hypergraph of degree
$1$ has ever been found, across $1{,}060{,}000$ instances on $14$ vertices and $23{,}542$
on $16$ vertices, and on $\approx160$ hand-built structurally distinct cores. Conversely
every degree-$1$ instance examined has a dominated vertex. Domination-free Cerberus
hypergraphs do exist, but only at degree $\ge2$; structured families reach degree $19$
(the dual affine plane $\mathrm{AG}(2,3)$) and $50$. The smallest degrees at which a
leaf-free domination-free core appears are small (degree $2$ upward), but never $1$.
